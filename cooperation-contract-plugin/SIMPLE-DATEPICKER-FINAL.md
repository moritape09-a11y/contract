
### JavaScript:
```javascript
$('#miladi_date').on('change', function() {
    const d = new persianDate(new Date($(this).val()));
    const jalali = d.format('YYYY/MM/DD');
    $('#contract_date').val(jalali);
    $('#jalali_result').html('📅 تاریخ شمسی: ' + jalali);
});
```
