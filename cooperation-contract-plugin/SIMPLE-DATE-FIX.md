## 📝 کد JavaScript جدید:

```javascript
// فقط 5 خط!
$('#contract_date').on('input', function() {
    var value = $(this).val().replace(/[^\d]/g, '');
    if (value.length >= 4) {
        var formatted = value.substring(0, 4);
        if (value.length >= 6) {
            formatted += '/' + value.substring(4, 6);
            if (value.length >= 8) {
                formatted += '/' + value.substring(6, 8);
            }
        }
        $(this).val(formatted);
    }
});
```
